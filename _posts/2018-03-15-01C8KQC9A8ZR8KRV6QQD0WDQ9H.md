                "node": true,

        },
        "parser": "babel-eslint",
        "parserOptions": {
                "sourceType": "module",
                "ecmaVersion": 6,
                "ecmaFeatures": {
                        "experimentalObjectRestSpread": true,
                        "jsx": true
                }
        },
        "rules": {
                "strict": [
                        0
                ],
                "react/jsx-uses-react": [
                        1
                ],
                "no-console": [
                        0
                ],
                "semi": [
                        2,
                        "always"
                ],
                "no-unused-vars": [
                        2
                ]
        }
}
